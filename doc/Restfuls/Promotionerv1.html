resource :promotioners do
        
        desc "Create a promotioner account"
        post do      
                authenticate_promotioner!
                present:p_token, token
        end

        desc "Login promotioner account"
        post '/login' do
                login_promotioner!
                promotioner = Promotioner.find_by(:phone_number => params[:username])
                promotioner_token = PromotionerToken.find_by(:promotioner_id => promotioner.id)
                promotioner_token = PromotionerToken.new if promotioner_token.nil?

                token = SecureRandom.uuid
                key = SecureRandom.urlsafe_base64
                promotioner_token.p_token = token
                promotioner_token.p_key = key
                promotioner_token.promotioner_id = promotioner.id
                error!("Server is not work well", 500) if !promotioner_token.save

                access_token = Hash.new
                access_token["token"] = token
                access_token["key"] = key
                present access_token
        end


end
